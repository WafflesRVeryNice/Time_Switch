# Time Switch Settings

# Room name format
Time_Switch_Setting_RoomNameFormat_Name= Enabled Room Name Format

Time_Switch_Setting_RoomNameFormat_Option_off= off
Time_Switch_Setting_RoomNameFormat_Option_Format1= simple
Time_Switch_Setting_RoomNameFormat_Option_Format2= clean

Time_Switch_Setting_RoomNameFormat_Description_Menu_off= 
user setting - will be overridden by maps
check time switch wiki for detailed info on room name formats

Time_Switch_Setting_RoomNameFormat_Description_Menu_Format1= 
user setting - will be overridden by maps
uses format (timeline)(identifier)(name)

Time_Switch_Setting_RoomNameFormat_Description_Menu_Format2= 
user setting - will be overridden by maps
uses format (identifier)(name)(timeline)

Time_Switch_Setting_RoomNameFormat_Description_InGame_off= 
map setting - may break map
check time switch wiki for detailed info on room name formats

Time_Switch_Setting_RoomNameFormat_Description_InGame_Format1= 
map setting - may break map
uses format (timeline)(identifier)(name)

Time_Switch_Setting_RoomNameFormat_Description_InGame_Format2= 
map setting - may break map
uses format (identifier)(name)(timeline)


# Timeline Type selection (legacy timelines)
Time_Switch_Setting_LegacyTimelines_Name= Timelines

Time_Switch_Setting_LegacyTimelines_Option_auto= Auto Detection
Time_Switch_Setting_LegacyTimelines_Option_legacySaveFile= Save to File
Time_Switch_Setting_LegacyTimelines_Option_legacySaveSession= Save to Session

Time_Switch_Setting_LegacyTimelines_Description_auto= 
user setting - will be overridden by maps
detects characters used for timelines, using "_" and "-" disables time switch in a room
Time_Switch_Setting_LegacyTimelines_Description_legacySaveFile= 
user setting - will be overridden by maps
uses "a" and "b" for timelines, saved to save-file (resets on save deletion)
Time_Switch_Setting_LegacyTimelines_Description_legacySaveSession= 
user setting - will be overridden by maps
uses "a" and "b" for timelines, saved to session (resets on exiting map except when using save & quit)
